      let cart = [];

      function formatPrice(num) {
        return num.toLocaleString("de-DE", {
          minimumFractionDigits: 2,
          maximumFractionDigits: 2,
        });
      }

      const productList = document.getElementById("product-list");
      for (let key in CONFIG.produkte) {
        const p = CONFIG.produkte[key];
        const variantKeys = Object.keys(p.variants);
        const hideVariantSelect =
          variantKeys.length === 1 && variantKeys[0] === "Standard";

        productList.innerHTML += `
            <div class="product-card p-5 bg-white border border-slate-100 rounded-2xl flex justify-between items-center transition-all duration-300">
                <div class="flex-1">
                    <h4 class="font-semibold text-slate-800">${p.name}</h4>
                    <div class="flex gap-2 mt-2">
                        <select id="type-${key}" onchange="updateSizes('${key}')" 
                            class="${hideVariantSelect ? "hidden" : ""} font-medium bg-slate-100 border-none rounded-lg px-2 py-1 outline-none cursor-pointer">
                            ${variantKeys.map((v) => `<option value="${v}">${v}</option>`).join("")}
                        </select>
                        <select id="size-${key}" class="font-medium bg-slate-100 border-none rounded-lg px-2 py-1 outline-none cursor-pointer"></select>
                    </div>
                </div>
                <button onclick="addToCart('${key}')" class="bg-blue-50 text-blue-600 px-5 py-2 rounded-xl text-sm font-semibold hover:bg-blue-600 hover:text-white transition-colors">Hinzufügen</button>
            </div>
        `;
        updateSizes(key);
      }

      function updateSizes(key) {
        const type = document.getElementById(`type-${key}`).value;
        const sizeSelect = document.getElementById(`size-${key}`);
        const variantData = CONFIG.produkte[key].variants[type];

        sizeSelect.innerHTML = variantData.groessen
          .map((g, index) => {
            const isObject = typeof g === "object";
            const label = isObject ? g.label : g;
            const preis = isObject ? g.preis : variantData.preis;
            return `<option value="${index}">${label} - ${formatPrice(preis)} €</option>`;
          })
          .join("");
      }

      function addToCart(key) {
        const type = document.getElementById(`type-${key}`).value;
        const sizeIndex = document.getElementById(`size-${key}`).value;
        const pData = CONFIG.produkte[key];
        const variantData = pData.variants[type];
        const g = variantData.groessen[sizeIndex];

        const sizeLabel = typeof g === "object" ? g.label : g;
        const preis = typeof g === "object" ? g.preis : variantData.preis;

        cart.push({
          id: Date.now(),
          displayName: pData.name,
          size: sizeLabel,
          preis,
        });
        renderCart();
      }

      function removeFromCart(id) {
        cart = cart.filter((item) => item.id !== id);
        renderCart();
      }

      function renderCart() {
        const list = document.getElementById("cart-list");
        const total = cart.reduce((sum, item) => sum + item.preis, 0);
        document.getElementById("cart-count").innerText = cart.length;
        if (cart.length === 0) {
          list.innerHTML = `<li class="py-4 text-slate-400 text-sm italic text-center">Noch keine Artikel ausgewählt</li>`;
        } else {
          list.innerHTML = cart
            .map(
              (item) => `
            <li class="flex justify-between items-center py-3 animate-in fade-in slide-in-from-right-2">
                <div>
                    <p class="text-sm font-medium text-slate-800">${item.displayName}</p>
                    <p class="text-[11px] text-slate-400">${item.size} • ${formatPrice(item.preis)} €</p>
                </div>
                <button onclick="removeFromCart(${item.id})" class="text-slate-300 hover:text-red-500 transition-colors">
                    <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path d="M6 18L18 6M6 6l12 12"></path></svg>
                </button>
            </li>
          `,
            )
            .join("");
        }
        document.getElementById("total-price").innerText = formatPrice(total);
      }

      async function handleCheckout(event) {
        event.preventDefault();
        if (cart.length === 0) return alert("Warenkorb ist leer!");
        const errorBox = document.getElementById("error-message");
        const submitBtn = document.getElementById("submit-btn");
        errorBox.classList.add("hidden");

        const payload = {
          password: document.getElementById("order-password").value,
          firstName: document.getElementById("cust-first-name").value,
          lastName: document.getElementById("cust-last-name").value,
          email: document.getElementById("cust-email").value,
          total: document.getElementById("total-price").innerText,
          produkte: {},
        };

        cart.forEach((item) => {
          if (!payload.produkte[item.displayName])
            payload.produkte[item.displayName] = [];
          payload.produkte[item.displayName].push(item.size);
        });

        submitBtn.disabled = true;
        submitBtn.innerText = "Wird gesendet...";

        try {
          const response = await fetch(CONFIG.scriptURL, {
            method: "POST",
            body: JSON.stringify(payload),
          });
          const result = await response.json();
          if (result.status === "success") showSuccess(payload, result.orderId);
          else throw new Error(result.message);
        } catch (e) {
          errorBox.innerText = "Fehler: " + e.message;
          errorBox.classList.remove("hidden");
        } finally {
          submitBtn.disabled = false;
          submitBtn.innerText = "Jetzt bestellen";
        }
      }

      function showSuccess(data, orderId) {
        document.getElementById("shop-view").classList.add("hidden");
        document.getElementById("success-view").classList.remove("hidden");
        window.scrollTo(0, 0);
        const summary = document.getElementById("order-summary");
        let itemsHtml = Object.entries(data.produkte)
          .map(
            ([name, sizes]) =>
              `<div class="flex justify-between text-sm"><span>${name}</span> <span class="text-slate-500">${sizes.join(", ")}</span></div>`,
          )
          .join("");
        summary.innerHTML = `
            <div class="space-y-1 pb-4 border-b border-slate-100">
                <p class="text-xs uppercase tracking-wider text-slate-400 font-bold">Besteller</p>
                <p class="font-medium">${data.firstName} ${data.lastName}</p>
                <p class="text-sm text-slate-500">${data.email}</p>
            </div>
            <div class="space-y-1 pb-4 border-b border-slate-100">
                <p class="text-xs uppercase tracking-wider text-slate-400 font-bold">Bestell-ID</p>
                <p class="font-medium">${orderId}</p>
            </div>
            <div class="space-y-2 pt-2">
                <p class="text-xs uppercase tracking-wider text-slate-400 font-bold mb-3">Positionen</p>
                ${itemsHtml}
            </div>
            <div class="pt-4 border-t border-slate-100 flex justify-between items-center">
                <span class="font-bold">Gesamtbetrag</span>
                <span class="text-lg font-bold text-blue-600">${data.total} €</span>
            </div>
        `;
